# Commands for checking out the psi software repositories
# By using svn checkout, the packages are a live repository (i.e. updatable/commitable)

# Check out the external dependencies repository
svn co https://www.predsci.com/svn_ext_deps/trunk/ psi_ext_deps

# Check out the psi tools repository
svn co https://www.predsci.com/svn_tools/trunk/ psi_tools

# Follow included instructions to install.
